1.3.0 发布注意

email 变小写

```js
// 用户中心
// 排除大小写重复email 4个
// Stephanieliebert1926@gmail.com => deprecated#Stephanieliebert1926@gmail.com
// Spdorning@gmail.com => deprecated#Spdorning@gmail.com
// Prestontrolson@yahoo.com => deprecated#Prestontrolson@yahoo.com
// Isabellepnock@gmail.com => deprecated#Isabellepnock@gmail.com
db.t_anzir_account.updateMany(
    {
        email:
            {
                $in: [
                    "Stephanieliebert1926@gmail.com", "Spdorning@gmail.com", "Prestontrolson@yahoo.com","Isabellepnock@gmail.com"
                ]
            }
    },
    [{
        $set: {
            email: {$set: "deprecated#"+"$email"}  // 将email字段的值转换为小写
        }
    }]
);

// 用户中心
// 修改所有email统一为小写
db.t_anzir_account.updateMany(
  {},  // 过滤条件，空表示匹配所有文档
  [{
    $set: {
      email: { $toLower: "$email" }  // 将email字段的值转换为小写
    }
  }]
);

// 通知中心
```
